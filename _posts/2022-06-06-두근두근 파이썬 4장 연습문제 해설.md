```python
#7번
c_list = []
color = input("색상#1을 입력하시오 : ")
c_list.append(color)
color = input("색상#2을 입력하시오 : ")
c_list.append(color)
color = input("색상#3을 입력하시오 : ")
c_list.append(color)

import turtle
t = turtle.Turtle()
t.shape("turtle")

t.fillcolor(str(c_list[0]))
t.begin_fill()
t.circle(50)
t.end_fill()
t.penup()
t.fd(100)
t.pendown()
t.fillcolor(str(c_list[1]))
t.begin_fill()
t.circle(50)
t.end_fill()
t.penup()
t.fd(100)
t.pendown()
t.fillcolor(str(c_list[2]))
t.begin_fill()
t.circle(50)
t.end_fill()
```
