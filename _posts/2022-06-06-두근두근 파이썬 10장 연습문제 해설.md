```python
#두근두근 파이썬 10장 2번
from tkinter import *

window = Tk()

total=0

def add():
    global total
    num = int(e1.get())
    total += num
    l2['text'] = str(total)

def abst():
    global total
    num = int(e1.get())
    total -= num
    l2['text'] = str(total)

def clear():
    global total
    total = 0
    l2['text'] = str(0)
    e1.delete(0,END)

l1 = Label(window,text ="현재 합계:")
l2 = Label(window,text="0")
l1.grid(row=0,column=0)
l2.grid(row=0,column=1,columnspan=2)

e1 = Entry(window)
e1.grid(row=1,column=0,columnspan=3)

b1 = Button(window,text="더하기(+)",command=add)
b2 = Button(window,text="빼기(-)",command=abst)
b3 = Button(window,text="초기화",command=clear)
b1.grid(row=2,column=0)
b2.grid(row=2,column=1)
b3.grid(row=2,column=2)

window.mainloop()

```

```python
#두근두근 파이썬 10장 3번

from tkinter import *
import random

window = Tk()

num1 = random.randint(1,100)

def test():
    global num1
    num2 = int(e1.get())
    if num1 == num2:
        l1['text'] = "정답입니다!"
    elif num1 > num2:
        l1['text'] = "너무 낮아요!"
    else :
        l1['text'] = "너무 높아요!"

def clear():
    global num1
    num1 = random.randint(1,100)
    l1['text'] = "1-100까지 입력"
    e1.delete(0,END)

l1 = Label(window, text="1-100까지 입력")
l1.grid(row=0, column=0,columnspan=2)

e1 = Entry(window)
e1.grid(row=1,column=0,columnspan=2)

b1 = Button(window,text="숫자를 입력",command=test)
b2 = Button(window,text="게임을 다시 실행",command=clear)
b1.grid(row=2,column=0)
b2.grid(row=2,column=1)

window.mainloop()
```
