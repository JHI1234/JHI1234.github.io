```python
#2번
year = 0
balance = 1000

while balance <= 2000:
    year = year + 1
    interest = balance * 0.07
    balance = balance + interest
print(year,"년이 걸립니다.")

#2번 풀이 :
#'2000만원보다 작거나 같을 때까지'라는 조건이 참일 때까지 실행문이 반복되어야 한다.
#따라서 관계연산자가 잘못되었으므로 '>='를 '<='로 수정하여야 한다.
```

```python
#3번
n = 1234  
sum = 0   
while n > 0 :             # 변수 n의 값이 0보다 클 때까지 실행문을 반복한다.
    digit = n % 10        # digit = 4
    sum = sum + digit     # sum = 4
    n = n // 10           # n = 123
print(sum)                # 4 + 3 + 2 + 1 = 10 → sum의 값은 10이다.
```

```python
#11번
import turtle
t = turtle.Turtle()
t.shape("turtle")
t.color("red","yellow")
t.begin_fill()
while True:
    t.fd(200)
    t.left(170)
    if abs(t.pos()) < 1:   
        break
t.end_fill()

# abs(): 절댓값을 구하는 함수
# t.pos(): 터틀의 현재 위치를 알려주는 함수
# - 반복문 분석
#   → 200 전진 후 좌측 170도 회전을 좌표의 절댓값이 1보다 작을 때까지 반복
#   → 즉, 터틀이 (0,0)에 위치할 때 반복을 종료
#
```
