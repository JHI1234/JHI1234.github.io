## 두근두근 파이썬 3장 연습문제 풀이

```python
#두근두근 파이썬 3장 연습문제 풀이

#1번
x = int(input("첫 번째 정수를 입력하시오: "))
y = int(input("두 번째 정수를 입력하시오: "))
print("x:", x)
print("y:",y)
print("두수의 합: ", x + y)
print("두수의 차: ", x - y)
print("두수의 곱: ", x * y)
print("두수의 평균: ", (x + y) / 2)
print("큰수: ",max(x, y))
print("작은수: ",min(x, y))
```

```python
#2번
r = int(input("원기둥의 반지름 길이를 입력하시오: "))
h = int(input("원기둥의 높이를 입력하시오: "))

vol = r ** 2 * h * 3.141592

print("원기둥의 부피: ", round(vol,2))
```

#3번

number = int(input("정수를 입력하시오: "))

y = (number // 100) % 10
z = (number // 1000) % 10
tot = w + x + y + z

print("자리수의 합:", tot)
